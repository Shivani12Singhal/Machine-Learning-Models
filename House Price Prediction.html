# outlier treatment for bedrooms
Q1 = housingDS.bedrooms.quantile(0.25)
Q3 = housingDS.bedrooms.quantile(0.75)
IQR = Q3 - Q1
housingDS = housingDS[(housingDS.bedrooms >= Q1 - 1.5*IQR) & (housingDS.bedrooms <= Q3 + 1.5*IQR)]
